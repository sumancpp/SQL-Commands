SELECT * FROM NEWBANK;

SELECT 
COUNT(*) AS Total_Records,
MAX(asset) AS Max_Asset,
MIN(asset) AS Min_Asset,
AVG(asset) AS Avg_Asset,
SUM(asset) AS Total_Asset
FROM NEWBANK;

SELECT city, SUM(asset) AS City_Asset FROM NEWBANK GROUP BY city;
SELECT city, SUM(asset) AS City_Asset FROM NEWBANK GROUP BY city having sum (asset) > 60000;

SELECT AVG(asset) AS Kolkata_Avg_Asset FROM NEWBANK WHERE city = 'Kolkata';

SELECT branchname, asset FROM NEWBANK WHERE asset > (SELECT AVG(asset) FROM NEWBANK);//sub query
